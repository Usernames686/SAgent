```typescript
// code-sandbox.service.ts
import * as ivm from 'isolated-vm';

@Injectable()
export class CodeSandboxService {
  /**
   * 在隔离环境中运行学生代码
   * 安全约束：
   * - 内存限制 128MB
   * - 超时 5s
   * - 禁止文件系统/网络访问
   * - 返回 stdout + stderr
   */
  async runCode(code: string, input?: string): Promise<RunResult> {
    const isolate = new ivm.Isolate({ memoryLimit: 128 });
    const context = isolate.createContextSync();
    const jail = context.global;

    // 注入 console.log 捕获
    jail.setSync('global', jail.derefInto());
    jail.setSync('console', this.createConsoleProxy(context));

    // 注入 stdin
    if (input) {
      jail.setSync('__stdin', input);
    }

    try {
      const script = isolate.compileScriptSync(code, { timeout: 5000 });
      script.runSync(context, { timeout: 5000 });

      return {
        success: true,
        stdout: this.capturedOutput,
        stderr: '',
        executionTime: 0,
      };
    } catch (error) {
      return {
        success: false,
        stdout: this.capturedOutput,
        stderr: error.message,
        executionTime: 0,
      };
    } finally {
      isolate.dispose();
    }
  }
}
```

### Step 23：实现 4 级验证评分算法

| 属性 | 值 |
|------|-----|

```typescript
// exercise-scoring.service.ts
@Injectable()
export class ExerciseScoringService {
  constructor(
    private readonly sandboxService: CodeSandboxService,
    private readonly testRunnerService: TestRunnerService,
    private readonly aiReviewEngine: AiReviewEngine,
  ) {}

  async scoreExercise(
    code: string,
    exerciseData: ExerciseData,
    hintPenalty: number = 0,
  ): Promise<ExerciseScore> {
    // Level 1: 正则匹配
    const level1Score = this.scoreByPattern(code, exerciseData.checks);

    // Level 2: 运行验证
    let level2Score = 0;
    if (exerciseData.runtimeChecks) {
      const results = await this.scoreByRuntime(code, exerciseData.runtimeChecks);
      level2Score = results.score;
    }

    // Level 3: 单元测试
    let level3Score = 0;
    if (exerciseData.unitTests) {
      const results = await this.scoreByTests(code, exerciseData.unitTests);
      level3Score = results.score;
    }

    // Level 4: AI 评审
    let level4Score = 0;
    if (exerciseData.aiReview) {
      const report = await this.aiReviewEngine.reviewCode(
        code, exerciseData.aiReview.rubric, exerciseData.aiReview.dimensions
      );
      level4Score = Object.values(report.scores).reduce((a, b) => a + b, 0) / Object.keys(report.scores).length;
    }

    // 综合评分（根据难度调整权重）
    const weights = this.getWeightsByDifficulty(exerciseData.difficulty);
    const totalScore = Math.max(0,
      level1Score * weights[0] +
      level2Score * weights[1] +
      level3Score * weights[2] +
      level4Score * weights[3] -
      hintPenalty * 100
    );

    return { level1Score, level2Score, level3Score, level4Score, totalScore, details: {}, hintPenalty };
  }

  /** 难度 → 权重映射 */
  private getWeightsByDifficulty(difficulty: number): [number, number, number, number] {
    if (difficulty <= 2) return [0.3, 0.5, 0.1, 0.1]; // 入门：偏重正则+运行
    if (difficulty === 3) return [0.2, 0.3, 0.3, 0.2]; // 中级：均衡
    return [0.1, 0.2, 0.4, 0.3];                        // 高级：偏重测试+AI评审
  }
}
```

### Step 37：间隔重复算法实现（SM-2）

```typescript
// spaced-repetition.service.ts — 基于 SM-2 算法
@Injectable()
export class SpacedRepetitionService {
  /**
   * 计算下次复习时间
   * @param quality 回忆质量 0-5（0=完全忘记，5=完美回忆）
   * @param easeFactor 当前难度因子
   * @param interval 当前间隔（天）
   */
  calculateNextReview(
    quality: number,     // 0-5
    easeFactor: number,  // 默认 2.5
    interval: number,    // 默认 1
  ): { nextInterval: number; nextEaseFactor: number; nextReviewAt: Date } {
    // SM-2 核心算法
    let newEaseFactor = easeFactor + (0.1 - (5 - quality) * (0.08 + (5 - quality) * 0.02));
    newEaseFactor = Math.max(1.3, newEaseFactor);

    let newInterval: number;
    if (quality < 3) {
      newInterval = 1; // 回忆失败，重置为1天
    } else if (interval === 1) {
      newInterval = 6;
    } else {
      newInterval = Math.round(interval * newEaseFactor);
    }

    const nextReviewAt = new Date();
    nextReviewAt.setDate(nextReviewAt.getDate() + newInterval);

    return { nextInterval: newInterval, nextEaseFactor: newEaseFactor, nextReviewAt };
  }

  /** 获取今日需要复习的知识点 */
  async getDueReviews(userId: string): Promise<LearningProgress[]> {
    return this.progressRepo.find({
      where: {
        userId,
        nextReviewAt: LessThanOrEqual(new Date()),
        status: In(['passed', 'mastered']),
      },
      order: { nextReviewAt: 'ASC' },
    });
  }
}
```

**具体操作**：

```typescript
@Injectable()
export class LearningAdvisorService {
  /**
   * 生成个性化学习建议
   * 综合考虑：当前进度、错题模式、间隔重复、知识点依赖
   */
  async generateAdvice(userId: string): Promise<LearningAdvice[]> {
    const advice: LearningAdvice[] = [];

    // 1. 间隔重复提醒
    const dueReviews = await this.spacedRepetition.getDueReviews(userId);
    if (dueReviews.length > 0) {
      advice.push({
        type: 'review',
        priority: 'high',
        title: `你有 ${dueReviews.length} 个知识点需要复习`,
        items: dueReviews.map(r => r.nodeId),
        action: 'spaced-repetition',
      });
    }

    // 2. 前置知识未完成 → 建议先学前置
    const blockedPoints = await this.findBlockedKnowledge(userId);
    if (blockedPoints.length > 0) {
      advice.push({
        type: 'prerequisite',
        priority: 'high',
        title: '先完成前置知识，解锁更多内容',
        items: blockedPoints,
        action: 'learn-prerequisite',
      });
    }

    // 3. 错题较多 → 建议回顾
    const errorHotspots = await this.errorReview.getErrorHotspots(userId);
    if (errorHotspots.length > 0) {
      advice.push({
        type: 'error-review',
        priority: 'medium',
        title: '这些知识点错误率较高，建议回顾',
        items: errorHotspots,
        action: 'error-review',
      });
    }

    // 4. 学习连续性 → 建议继续当前模块
    const currentModule = await this.findCurrentModule(userId);
    if (currentModule) {
      advice.push({
        type: 'continue',
        priority: 'medium',
        title: `继续学习 ${currentModule.name}`,
        items: [currentModule.nextKnowledgeId],
        action: 'continue-learning',
      });
    }

    return advice.sort((a, b) => this.priorityWeight(b.priority) - this.priorityWeight(a.priority));
  }
}
```

**验证方法**：根据不同学习状态，返回合理的建议列表
